# secret-paths.txt — pure-secret credential stores Claude must never READ.
#
# Sourced by hooks/guard-secret-paths.sh (file tools) and hooks/guard-bash.sh
# (Bash commands). One extended-regex (ERE) per line; matched case-insensitively
# against a tool's file_path/pattern/path AND against bash command text.
# Blank lines and lines starting with # are ignored.
#
# SCOPE = pure-secret stores only (balanced calibration): things Claude never
# legitimately needs to read. Dual-use files (.env, .npmrc) are deliberately NOT
# listed here — they're protected at the *exit* layer by githooks/pre-commit +
# scan-tree.sh instead, so reading them for config stays frictionless.
#
# Do NOT add ~/.zshrc / ~/.p10k.zsh — shell-config EDITS are intentionally allowed.

(^|/)\.secrets($|[^./A-Za-z0-9_-])
(^|/)\.ssh($|[^.A-Za-z0-9_-])
(^|/)\.aws($|[^.A-Za-z0-9_-])
(^|/)\.gnupg($|[^.A-Za-z0-9_-])
(^|/)\.config/gcloud($|[^.A-Za-z0-9_-])
(^|/)\.kube($|[^.A-Za-z0-9_-])
(^|/)\.config/gh($|[^.A-Za-z0-9_-])
(^|/)\.git-credentials($|[^A-Za-z0-9])
/Library/Keychains($|[^.A-Za-z0-9_-])
\.pem($|[^A-Za-z0-9])
\.key($|[^A-Za-z0-9])
\.(jks|keystore|p12|mobileprovision)($|[^A-Za-z0-9])
service-account[^/]*\.json
gsheets_key
(^|/)\.mcp\.local\.json
